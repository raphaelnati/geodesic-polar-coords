Program Use¶

The only two neccesary things to adjust to get a base run of this program is adding a name to "filename" and changing the value "Umax"

The filename will be the name of the desired obj file without the file extension ".obj" typed at the end

for example if using "bunny.obj", filename will just be the word "bunny"

 

Umax is located in the cell under the markdown cell titled "Algorithm Values"

Umax will need to be adjusted per mesh as there is no standard size for obj inputs

Umax will want to be chosen so that a decently small size of the mesh is calculated for parameterization, otherwise the parameterization might create incorrect values.

The following functions apply the geodesic polar coordinate algorithm on a triangular mesh based on a specific starting point on the mesh

#compute distance
def computeDistance(vi,vj,vk,Uj,Uk,s,j,k):
    #uijkloc is 0 if xj and xk > 0, 1 if updated from j, 2 if updated from k
    uijkloc = 0
    
    ej = vj - vi
    ek = vk - vi
    ekj = vk - vj
    ekj2 = np.dot(ekj,ekj)
    
    A = np.linalg.norm(np.cross(ej,ek))
    #H = np.sqrt((ekj2 - pow(Uj - Uk,2) )* (pow(Uj + Uk,2) - ekj2))
    
    
    
    #more stable version of heron's formula
    hset = [Uk,Uj,np.linalg.norm(ekj)]
    hset.sort(reverse=True)
    hinner = (hset[0] + (hset[1] + hset[2]))*(hset[2]- (hset[0] -hset[1]))*(hset[2]+ (hset[0] -hset[1]))*(hset[0] + (hset[1] -hset[2]))
    
    #this check is for if any pair of Uj, Uk or ejk add up to the third entry, in which case it will be a degenerate triangle
    if hinner <= 0:
        return float("inf"), 0, np.array([0,0,0])
        #hinner = 0
    
    H= np.sqrt(hinner)
    
    #calculate xj and xk, values for s point projection
    xj = (A * (ekj2 + pow(Uk,2) - pow(Uj,2)) + np.dot(ek, ekj) * H)/(2 * A * ekj2)
    xk = (A * (ekj2 + pow(Uj,2) - pow(Uk,2)) - np.dot(ej, ekj) * H)/(2 * A * ekj2)
    
    Uijk = 0
    #step to find min value of Uijk
    if (xj > 0 and xk > 0):
        Uijk = np.linalg.norm(xj * ej + xk * ek)
    else:
        tempuk = Uk + np.linalg.norm(ek)
        tempuj = Uj + np.linalg.norm(ej)
        if tempuj < tempuk:
            uijkloc = 1
            Uijk = tempuj
        else:
            uijkloc = 2
            Uijk = tempuk
    #calculate s projection point
    sp = vi + xj * ej + xk * ek
    return Uijk, uijkloc, sp

def computeAngle(vi,vj,vk,sp, thetaj, thetak, uijkloc, i):
    #comput angle for new polar co-ordinate
    
    
    #return if calculated from previously calculated coordinate
    if uijkloc == 1:
        return thetaj% (2*math.pi)
    elif uijkloc == 2:
        return thetak% (2*math.pi)
    
    
    #prevent 360 degree comparisons
    thetaj2 = thetaj
    thetak2 = thetak
    if (thetaj > thetak):
        diffang = thetaj - thetak
        if (diffang >= math.pi):
            thetaj2 -= 2*math.pi
    if (thetaj < thetak):
        diffang = thetak - thetaj
        if (diffang >= math.pi):
            thetak2 -= 2*math.pi
    
    
    #get angles between Uj and Uijk, and Uj and Uk from the projection point sp
    vks = vk- sp
    vjs = vj - sp
    vis = vi - sp
    vks = vks / np.linalg.norm(vks)
    vjs = vjs / np.linalg.norm(vjs)
    vis = vis / np.linalg.norm(vis)
    pkjin = np.dot(vks,vjs)/np.linalg.norm(vks) * np.linalg.norm(vjs)
    pijin = np.dot(vis,vjs)/np.linalg.norm(vis) * np.linalg.norm(vjs)
    
    
    if (pkjin < -1):
        pkjin = -1
    elif(pkjin > 1):
        pkjin = 1
    if (pijin < -1):
        pijin = -1
    elif(pijin > 1):
        pijin = 1
    #get angle percentage alpha that phi ij is of phi kj
    phikj = math.acos(pkjin)
    phiij = math.acos(pijin)
    alpha = phiij / phikj
    #get final angle
    finang = ((1 - alpha)*thetaj2 + alpha * thetak2) % (2*math.pi)
    return finang

def projectVec(normal, vec):
    #project a vector onto a plane given the plane's normal
    projection = vec - (np.dot(vec,normal) / pow(np.linalg.norm(normal),2))*normal
    return projection
def getAngle(normal, x, vec):
    #get angle between to vectors with respect to a plane given the planes normal
    #x is the reference vector
    angle = math.atan2((np.dot(normal, np.cross(x,vec))),np.dot(x,vec))
    angle = (angle + 2*math.pi) % (2*math.pi)
    return angle

def initializeNeighbourhood(s):
    #U is the distance of the vertex
    U = []
    #theta is the angle of the vertex from the reference vector
    theta = []
    #checked is an array of which vertices have been checked in the algorithm
    checked = []
    #candidates is a heap that stores the closest distance vertex and the reference vertex
    candidates = []
    #intialize arrays
    for i in range(0,v.size,3):
        U.append(1000000)
        theta.append(0)
        checked.append(False)
    
    #x is the angle reference vector, s is the index of main vertex
    x = np.array([])
    xind = 0
    #initial point angle and distance will be 0
    U[s] = 0
    theta[s] = 0
    checked[s]= True
    #initialize all adjacent vertices to initial point
    for i in vertexList[s]:
        temp = getArr(v,i) - getArr(v,s)
        #intialize first checked vertex as reference angle
        if x.size == 0:
            xind = i
            x = projectVec(getArr(vn,s), temp)
        #get distance for neighbouring vertex
        U[i] = np.linalg.norm(temp)
        #push distance to vertex as well as vertex number
        hq.heappush(candidates,(U[i],i))
        #get angle projected onto plane from main vertex's normal
        theta[i] = getAngle(getArr(vn,s), x, temp)
        checked[i] = True
    return x, U, theta, checked, candidates, xind

def geodesicAlgo(s,Umax,eps):
    x, U, theta, checked, candidates, xind = initializeNeighbourhood(s)
    #keep going while candidates list is not empty
    while len(candidates) >0:
        #pull closest candidate's distance and index
        newpop = hq.heappop(candidates)
        Uj = newpop[0]
        j = newpop[1]
        #for all vertices adjacent to j
        for i in vertexList[j]:
            #intersection of triangles containing i and j
            sameTriangles = triangleList[j] & triangleList[i]
            #pull triangles from set
            while len(sameTriangles) > 0:
                triArr = np.copy(getArr(f,sameTriangles.pop()))
                k = np.delete(triArr,np.where(triArr==i))
                #get third vertex and determine if a value for it has been calculated for it yet
                k = np.delete(k,np.where(k==j))[0]
                if checked[k] == True:
                    #run compute distance
                    newUi, uijkloc, sp = computeDistance(getArr(v,i),getArr(v,j),getArr(v,k),U[j],U[k],getArr(v,s),j,k)
                    #if new distance is smaller than calculated distance given some threshold
                    if (U[i] / newUi) > (1 + eps):
                        #set new closer distance and get angle
                        U[i] = newUi 
                        checked[i] = True
                        theta[i] = computeAngle(getArr(v,i),getArr(v,j),getArr(v,k),sp,theta[j],theta[k],uijkloc,i)
                        #theta[i] = computeAngle(getArr(v,i),getArr(v,j),getArr(v,k),getArr(v,s),theta[j],theta[k],uijkloc,i)
                        #if distance is inside range, push to candidates
                        if newUi < Umax:
                            hq.heappush(candidates, (U[i],i))
    return U, theta, checked, xind

#cartesian coordinates to barycentric coordinates, given 3 vertices and point p
def cartToBary(vi,vj,vk,p):
    area = np.linalg.norm(np.cross(vj-vi, vk-vi)) * 0.5
    b = (np.linalg.norm(np.cross(p - vi, vk- vi)) *0.5) / area
    a = (np.linalg.norm(np.cross(p - vk, vj- vk)) *0.5) / area
    #c = (np.linalg.norm(np.cross(p - vj, vi- vj)) *0.5) / area
    c = 1.0 - (a + b)
    return np.array([a,b,c])
#barycentric coordinates to cartesian coordinates, given 3 vertices and barycentric coordinates in an array abc
def baryToCart(abc,vi,vj,vk):
    return abc[0] * vi + abc[1] * vj + abc[2] * vk

#checks if point p is contained in triangle [vi,vj,vk]
def pointContained(vi,vj,vk,p):
    uj = vj - vi
    uk = vk - vi
    #print(np.linalg.det(np.vstack((uj,uk)).T))
    if (np.linalg.det(np.vstack((uj,uk)).T) == 0):
#         print("vi ", vi, " vj ", vj, " vk ", vk, " p ", p)
#         print(np.vstack((uj,uk)).T)
        return False
    a = np.linalg.inv(np.vstack((uj,uk)).T)
    xu = np.dot(a, (p - vi))
    if (xu[0] < 0 or xu[1] < 0):
        return False
    if (np.sum(xu) > 1):
        return False
    return True

def laplacian(beta):
    #object vertices to store laplacian version of vertices
    v2 = np.copy(v)
    #cartesian coordinates to store the laplacian version of coordinates
    xcart2 = np.copy(xcart)
    ycart2 = np.copy(ycart)
    for i in range (0, len(checked)):
        #check if vertex has been claculated
        if checked[i]:
            adj = list(vertexList[i])
            #check if all adjacent vertices have been checked as well
            if (listChecked(adj) == True):
                #average surrounding vertices
                ipoint = np.array([xcart[i],ycart[i]])
                totx = 0
                toty = 0
                for vert in adj:
                    totx += xcart[vert] - ipoint[0]
                    toty += ycart[vert] - ipoint[1]
                totx = totx / len(adj) * beta
                toty = toty / len(adj) * beta
                newP = np.array([xcart[i] + totx, ycart[i] + toty])
                #get laplacian version of coordinate
                xcart2[i] = newP[0]
                ycart2[i] = newP[1]
                #find which triangle new point lies in
                for tris in list(triangleList[i]):
                    triInds = getArr(f,tris)
                    vi2= np.array([xcart[triInds[0]],ycart[triInds[0]]])
                    vj2 =np.array([xcart[triInds[1]],ycart[triInds[1]]])
                    vk2 =np.array([xcart[triInds[2]],ycart[triInds[2]]])
                    if pointContained(vi2,vj2,vk2,newP):
                        #if point is in current triangle, get barycentric coordinate in parameter mesh and displace object
                        #to that point
                        baryCoord = cartToBary(vi2,vj2,vk2,newP)
                        vi3 = getArr(v,triInds[0])
                        vj3 = getArr(v, triInds[1])
                        vk3 = getArr(v, triInds[2])
                        vpnew3 = baryToCart(baryCoord, vi3,vj3,vk3)
                        v2[i*3:i*3 + 3] = vpnew3
    return v2, xcart2, ycart2

def resample(pointdimension):
    #space sampled over is a square that is half the size of Umax
    xdim = np.linspace(-1 * (Umax / 2), Umax / 2, pointdimension)
    ydim = np.linspace(-1 * (Umax / 2), Umax / 2, pointdimension)
